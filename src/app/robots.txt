				"/legal/dmca",
				"/settings/",
				"/notifications",
				"/bookmarks",

				"/forgot-password",
				"/reset-password",
				"/banned",
			],
		},
		sitemap: "https://glimpse.aditsuru.com/sitemap.xml",
	};
}
